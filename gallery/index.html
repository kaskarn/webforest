Code 
library(tabviz)

set.seed(42)
dashboard_data <- data.frame(
  department = c("Engineering", "Sales", "Marketing", "Operations", "Support"),
  headcount = c(85, 42, 28, 35, 22),
  satisfaction = c(4.2, 3.9, 4.1, 3.7, 4.4),
  budget_var = c(0.05, 0.12, -0.08, 0.02, -0.15),
  budget_lo = c(-0.02, 0.04, -0.15, -0.05, -0.22),
  budget_hi = c(0.12, 0.20, -0.01, 0.09, -0.08),
  response_dist = I(list(
    rnorm(80, 85, 10),
    rnorm(40, 78, 15),
    rnorm(25, 82, 12),
    rnorm(30, 75, 18),
    rnorm(20, 88, 8)
  )),
  trend = I(list(
    c(78, 80, 82, 84, 85, 86, 87, 88, 89, 90, 91, 92),
    c(70, 72, 74, 75, 76, 77, 78, 78, 79, 79, 80, 80),
    c(75, 77, 79, 80, 81, 82, 83, 84, 84, 85, 85, 86),
    c(68, 70, 71, 72, 73, 74, 74, 75, 75, 76, 76, 77),
    c(82, 84, 85, 86, 87, 88, 89, 90, 91, 92, 92, 93)
  ))
)

tabviz(dashboard_data, label = "department",
  columns = list(
    col_n("headcount", "FTEs"),
    viz_bar(
      viz_bar_effect("satisfaction"),
      header = "Satisfaction",
      axis_range = c(0, 5)
    ),
    viz_forest(
      point = "budget_var", lower = "budget_lo", upper = "budget_hi",
      null_value = 0,
      header = "Budget Var"
    ),
    viz_boxplot(
      viz_boxplot_effect(data = "response_dist"),
      header = "Performance",
      show_outliers = TRUE
    ),
    col_sparkline("trend", "12M Trend")
  ),
  # Row-level marker colors: green for positive, red for negative, gray for non-significant
  marker_color = ~ ifelse(budget_lo > 0, "#16a34a",
                          ifelse(budget_hi < 0, "#dc2626", "#64748b")),
  theme = web_theme_modern(),
  title = "Department Performance Dashboard"
)

## Quick Examples

### Clinical Trial Meta-Analysis

```{r}
#| echo: false
library(tabviz)
data(glp1_trials)

tabviz(
  glp1_trials,
  label = "study",
  group = "group",
  columns = list(
    col_text("drug", "Drug"),
    col_n("n"),
    viz_forest(point = "hr", lower = "lower", upper = "upper",
               scale = "log", null_value = 1,
               axis_label = "Hazard Ratio"),
    col_interval("HR (95% CI)", point = "hr", lower = "lower", upper = "upper"),
    col_pvalue("pvalue", "P")
  ),
  row_type = "row_type",
  row_bold = "row_bold",
  theme = web_theme_nature()
)
```

### Dashboard Table

```{r}
#| echo: false
library(tabviz)
data(nba_efficiency)

tabviz(
  nba_efficiency,
  label = "player",
  columns = list(
    col_text("team", "Team"),
    col_text("position", "Pos"),
    col_numeric("ppg", "PPG", decimals = 1),
    col_bar("per", "PER", max_value = 35),
    col_interval("PER (95% CI)", point = "per", lower = "per_lower", upper = "per_upper", decimals = 1)
  ),
  theme = web_theme_modern()
)
```
